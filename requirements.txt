# AURALINK audio engine dependencies.
# Apple Silicon required (MLX backend). Install into a Python 3.12 venv:
#   uv venv --python 3.12 && source .venv/bin/activate
#   uv pip install -r requirements.txt
# Then download the model once:
#   mrt models init
#   mrt checkpoints download mrt2_small.safetensors
magenta-rt[mlx]
numpy
sounddevice
soundfile

# Live Pulsoid heart rate (--pulsoid), e.g. an Apple Watch. Reads a Pulsoid
# access token from the PULSOID_TOKEN env var (see .env.example).
websocket-client

# Auto-loads a local .env file (e.g. PULSOID_TOKEN) at startup. See .env.example.
python-dotenv

# Web dashboard bridge (--serve): serves frontend/ and streams live heart-rate /
# tempo / zone state to the browser over a WebSocket. Audio stays local (Magenta
# via sounddevice); the browser is a telemetry + control surface only.
fastapi
uvicorn[standard]
